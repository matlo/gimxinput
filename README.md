# gimxinput

```
git clone https://github.com/matlo/gimxpoll.git
git clone https://github.com/matlo/gimxcommon.git

cd gimxhid
CPPFLAGS="-I../" make
cd ..
git clone https://github.com/matlo/gimxinput.git
cd gimxinput
CPPFLAGS="-I../" make
```
